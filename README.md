# Professional Data Science Pipeline
import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns

# Data Engineering & Preprocessing
def load_and_clean_data(filepath):
    """Load and preprocess Diwali sales data"""
    df = pd.read_csv(filepath)
    df.dropna(subset=['Amount'], inplace=True)
    df['State'] = df['State'].str.strip()
    return df

# Advanced Analytics Functions  
def analyze_revenue_by_category(df):
    """Revenue analysis by product category"""
    return df.groupby('Product_Category')['Amount'].sum().sort_values(ascending=False)

def demographic_analysis(df):
    """Customer demographic segmentation"""
    return df.groupby(['Gender', 'Age Group'])['Amount'].sum()

# Statistical Insights
df = load_and_clean_data('Diwali Sales Data.csv')
